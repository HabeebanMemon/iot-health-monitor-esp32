unsigned long lastEmailSentTime = 0;

void sendEmail(String subject, String messageBody) {
  unsigned long currentTime = millis();

  if (currentTime - lastEmailSentTime < 15000) {
    Serial.println("Email will be sent after 15 seconds.");
    return;
  }

  ESP_Mail_Session session;
  session.server.host_name = SMTP_server;
  session.server.port = SMTP_Port;
  session.login.email = sender_email;
  session.login.password = sender_password;

  SMTP_Message message;
  message.sender.name = "Patient Habiba Health Monitor Alert!!";
  message.sender.email = sender_email;
  message.subject = subject.c_str();
  message.addRecipient(Recipient_name, Recipient_email);

  String htmlMsg = "<div style=\"color:#000000;\"><h2>Health Monitor Alert</h2><p>" + messageBody + "</p></div>";
  message.html.content = htmlMsg.c_str();
  message.html.charSet = "us-ascii";

  if (!smtp.connect(&session)) {
    Serial.println("Error connecting to SMTP server: " + smtp.errorReason());
    return;
  }

  if (!MailClient.sendMail(&smtp, &message)) {
    Serial.println("Error sending Email: " + smtp.errorReason());
  } else {
    Serial.println("Email sent successfully!");
    lastEmailSentTime = currentTime;
  }
}

void setup() {

void loop() {
  if (wifiConnected) {
    Blynk.run();
  }

  if (pulseSensor.sawStartOfBeat()) {
    myBPM = pulseSensor.getBeatsPerMinute();
    
    if (myBPM > 40 && myBPM < 180) {
      Serial.print("BPM: ");
      Serial.println(myBPM);

      lcd.setCursor(0, 0);
      lcd.print("Heart Rt: ");
      lcd.print(myBPM);
      lcd.print(" BPM ");

      lcd.setCursor(0, 1);
      if (myBPM < normalMin) {
        lcd.print("Low HR Alert! ");
        digitalWrite(Buzzer, HIGH);
        Serial.println("Low Heart Rate!");
        sendEmail("Low Heart Rate Alert", "The patient's heart rate is dangerously low: " + String(myBPM) + " BPM.");
      } else if (myBPM > normalMax) {
        lcd.print("High HR Alert!");
        digitalWrite(Buzzer, HIGH);
        Serial.println("High Heart Rate!");
        sendEmail("High Heart Rate Alert", "The patient's heart rate is dangerously high: " + String(myBPM) + " BPM.");
      } else {
        lcd.print("HR Normal     ");
        digitalWrite(Buzzer, LOW);
        Serial.println("Normal Heart Rate.");
      }

      if (wifiConnected) {
        Blynk.virtualWrite(V3, myBPM);
      }
    }
  }

  int tempValue = analogRead(TempPin);
  temperature = (tempValue * 3.3 / 4095.0) * 100;

  char tempStr[6];
  dtostrf(temperature, 4, 1, tempStr);

  lcd.setCursor(0, 1);
  lcd.print("Room Tmp: ");
  lcd.print(tempStr);
  lcd.print(" C   ");

  if (wifiConnected) {
    Blynk.virtualWrite(V4, temperature);
  }

  delay(4000);
}
